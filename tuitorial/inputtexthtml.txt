        <div class="form-group">
            <label for="nameInput">Name:</label>
            <input type="text" id="nameInput" placeholder="Type your name here...">
        </div>
        
        <button id="submitButton">Submit</button>
        
        <p id="output"></p>
    </div>

    <script>
        // Get references to the HTML elements we will interact with
        const nameInput = document.getElementById('nameInput');
        const submitButton = document.getElementById('submitButton');
        const outputParagraph = document.getElementById('output');

        // Add a click event listener to the button
        submitButton.addEventListener('click', function() {
            // Get the string value from the input field
            // The .value property is used to retrieve what the user typed
            const inputValue = nameInput.value;

            // Check if the input is not empty
            if (inputValue.trim() !== "") {
                // Display the retrieved string in the output paragraph
                outputParagraph.textContent = `Hello, ${inputValue}!`;
            } else {
                // If the input is empty, show a message
                outputParagraph.textContent = "Please enter a name.";
            }
        });
    </script>
