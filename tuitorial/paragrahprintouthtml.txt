<body>

    <h1>Dynamically Adding a Paragraph</h1>

    <div id="container">
        <!-- New paragraphs will be added here by JavaScript -->
        <p>This is the initial paragraph.</p>
    </div>

    <button id="addParagraphButton">Add a New Paragraph</button>

    <script>
        // Get a reference to the container and the button
        const container = document.getElementById('container');
        const addButton = document.getElementById('addParagraphButton');

        let paragraphCount = 1;

        // Function to create and add a new paragraph
        function addNewParagraph() {
            // 1. Create a new paragraph element
            const newParagraph = document.createElement('p');
            
            // 2. Set the text content for the new paragraph
            paragraphCount++;
            newParagraph.textContent = `This is a new paragraph, number ${paragraphCount}.`;

            // 3. Append the new paragraph to the container
            container.appendChild(newParagraph);
        }

        // Add a click event listener to the button
        addButton.addEventListener('click', addNewParagraph);
    </script>
